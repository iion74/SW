  <script>
    const exploreBtn = document.getElementById('exploreBtn');
    const mainNav = document.getElementById('mainNav');
    const sections = Array.from(document.querySelectorAll('section[id]')).filter(s => s.id !== 'home');

    function showOnly(selector){
      sections.forEach(s => s.classList.add('hidden'));
      const el = document.querySelector(selector);
      if(el) el.classList.remove('hidden');
    }

    // 둘러보기 → 팀원 소개만
    exploreBtn.addEventListener('click', (e) => {
      e.preventDefault();
      mainNav.classList.remove('hidden');
      showOnly('#members');
      document.querySelector('#members').scrollIntoView({ behavior:'smooth', block:'start' });
    });

    // 네비게이션 클릭
    document.querySelectorAll('header nav a[data-target]').forEach(a=>{
      a.addEventListener('click', (e)=>{
        e.preventDefault();
        const t = a.getAttribute('data-target');
        if(t){ mainNav.classList.remove('hidden'); showOnly(t); document.querySelector(t).scrollIntoView({behavior:'smooth'}); }
      });
    });

    // 팀원 클릭 → 해당 멤버의 프로젝트 섹션 보여주기
    document.querySelectorAll('#members [data-member]').forEach(btn=>{
      btn.addEventListener('click', ()=>{
        const key = btn.getAttribute('data-member'); // S, C, N, U, AI
        showOnly(`#projects-${key}`);
        document.querySelector(`#projects-${key}`).scrollIntoView({ behavior:'smooth' });
      });
    });

    // 보기 버튼: 이미지 모달 (이벤트 위임)
    document.addEventListener('click', (e) => {
      const link = e.target.closest('.view-image');
      if(!link) return;
      e.preventDefault();
      const src = link.getAttribute('data-image');
      const modal = document.createElement('div');
      modal.className = 'modal';
      modal.innerHTML = `
        <div class="bg-white rounded-xl shadow-2xl p-3 md:p-4 max-w-4xl w-[92%]">
          <div class="text-right">
            <button id="closeModal" class="px-3 py-1 text-sm rounded-md bg-black text-blue-500">닫기</button>
          </div>
          <img src="${src}" alt="프로젝트 이미지" class="max-h-[78vh] w-auto mx-auto rounded-lg mt-2"/>
        </div>`;
      document.body.appendChild(modal);

      // 닫기: 배경 클릭 / 버튼 클릭 / ESC
      modal.addEventListener('click', (ev)=>{ if(ev.target === modal) modal.remove(); });
      modal.querySelector('#closeModal').addEventListener('click', ()=> modal.remove());
      window.addEventListener('keydown', function onEsc(ev){ if(ev.key === 'Escape'){ modal.remove(); window.removeEventListener('keydown', onEsc); }});
    });
  </script>
